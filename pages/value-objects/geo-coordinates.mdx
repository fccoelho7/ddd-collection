# Geo Coordinates

```ts copy
class GeoCoordinates {
  private latitude: number;
  private longitude: number;

  constructor(latitude: number, longitude: number) {
    if (
      latitude < -90 ||
      latitude > 90 ||
      longitude < -180 ||
      longitude > 180
    ) {
      throw new Error(
        "Invalid coordinates. Latitude should be between -90 and 90, and longitude should be between -180 and 180."
      );
    }
    this.latitude = latitude;
    this.longitude = longitude;
  }

  equals(other: GeoCoordinates): boolean {
    return (
      this.latitude === other.latitude && this.longitude === other.longitude
    );
  }

  toString(): string {
    return `Latitude: ${this.latitude}, Longitude: ${this.longitude}`;
  }
}

export { GeoCoordinates };
```

import { GeoCoordinates } from "./GeoCoordinates"; // Adjust the import path as per your project structure

describe("GeoCoordinates", () => {
  describe("constructor", () => {
    it("should create a GeoCoordinates instance with valid latitude and longitude", () => {
      const coordinates = new GeoCoordinates(37.7749, -122.4194); // San Francisco, CA
      expect(coordinates).toBeDefined();
    });

    it("should throw an error for latitude out of range", () => {
      expect(() => new GeoCoordinates(-91, -122.4194)).toThrow();
      expect(() => new GeoCoordinates(91, -122.4194)).toThrow();
    });

    it("should throw an error for longitude out of range", () => {
      expect(() => new GeoCoordinates(37.7749, -181)).toThrow();
      expect(() => new GeoCoordinates(37.7749, 181)).toThrow();
    });
  });

  describe("equals", () => {
    it("should return true when comparing two GeoCoordinates instances with the same latitude and longitude", () => {
      const coordinates1 = new GeoCoordinates(37.7749, -122.4194); // San Francisco, CA
      const coordinates2 = new GeoCoordinates(37.7749, -122.4194); // San Francisco, CA
      expect(coordinates1.equals(coordinates2)).toBe(true);
    });

    it("should return false when comparing two GeoCoordinates instances with different latitude or longitude", () => {
      const coordinates1 = new GeoCoordinates(37.7749, -122.4194); // San Francisco, CA
      const coordinates2 = new GeoCoordinates(40.7128, -74.006); // New York City, NY
      expect(coordinates1.equals(coordinates2)).toBe(false);
    });
  });

  describe("toString", () => {
    it("should return the string representation of the GeoCoordinates", () => {
      const coordinates = new GeoCoordinates(37.7749, -122.4194); // San Francisco, CA
      expect(coordinates.toString()).toBe(
        "Latitude: 37.7749, Longitude: -122.4194"
      );
    });
  });
});
